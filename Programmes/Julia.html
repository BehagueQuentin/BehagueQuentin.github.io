Ensemble de Julia

Une fractale de Julia est un ensemble obtenu en associant une suite à chaque point du plan complexe, les points de l'ensemble sont alors ceux dont la suite associée est bornée. Sur cette page, on peut choisir la valeur de \(c\) dans l'expression de la suite. À chaque point du plan \(z\) on associe la suite : \[(z_n) : \left\{\begin{array}[c]{} z_0 = z \\ \forall n \in \mathbb N, z_{n+1} = z_{n}^2 + c \end{array} \right.\] Ici les points du plan pour lesquels la suite est convergente sont coloriés en noir, la couleur des points dont la suite diverge varie en fonction de la vitesse de divergence.
Il est possible de télécharger la version python de ce code sur mon GitHub. Une implémentation du flocon de Koch y est aussi disponible.

import pygame
pygame.init()
ITERATION = 80
abcisseMIN, abcisseMAX, ordonneeMIN, ordonneeMAX = -1.75, 1.75, -1.75, 1.75 
LARGEUR, HAUTEUR = 400,400
fenetre = pygame.display.set_mode((LARGEUR,HAUTEUR))
pygame.display.set_caption("Ensemble de Julia")
for y in range(HAUTEUR):
    pygame.display.flip()
    for x in range(LARGEUR):
        xn = (x * (abcisseMAX - abcisseMIN) / LARGEUR + abcisseMIN)
        yn = (y * (ordonneeMIN - ordonneeMAX) / HAUTEUR + ordonneeMAX)
        cx = 0.3  # Re(c)
        cy = 0.5  # Im(c)
        n = 0
        while (xn * xn + yn * yn) < 4 and n < ITERATION:
            xn, yn = xn*xn - yn*yn +cx , 2*xn*yn+cy
            n += 1
        if n == ITERATION:
            fenetre.set_at((x, y), (0,0,0))
        else:
            fenetre.set_at((x, y),((5* n+10) % 256, (20 * n) % 256, (2 * n+90) % 256))     
pygame.display.flip()
ouvert= True
while ouvert:
    for event in pygame.event.get():
    if event.type == pygame.QUIT:
        ouvert = False
pygame.quit()

function drawJuliaSet(canvas, cReal, cImaginary, maxIterations) {
    const ctx = canvas.getContext('2d');
    const imageData = ctx.createImageData(canvas.width, canvas.height);

    for (let x = 0; x < canvas.width; x++) {
        for (let y = 0; y < canvas.height; y++) {
            const real = (x - canvas.width / 2) / (canvas.width / 2);
            const imag = (y - canvas.height / 2) / (canvas.height / 2);
            let zReal = real;
            let zImag = imag;
            let iteration = 0;
            while (iteration < maxIterations) {
                const zRealTemp = zReal * zReal - zImag * zImag + cReal;
                zImag = 2 * zReal * zImag + cImaginary;
                zReal = zRealTemp;
                if (zReal * zReal + zImag * zImag > 4) {
                    break;
                }
                iteration++;
            }
            const pixelIndex = (y * canvas.width + x) * 4;
            if(iteration === maxIterations){
                imageData.data[pixelIndex] = 0; 
                imageData.data[pixelIndex + 1] = 0; 
                imageData.data[pixelIndex + 2] = 0; 
                imageData.data[pixelIndex + 3] = 255;
            }
            else{
                const color =  255 - Math.floor(iteration / maxIterations * 255);
                imageData.data[pixelIndex] = (5*color+10)%256;
                imageData.data[pixelIndex + 1] = (20*color)%256;
                imageData.data[pixelIndex + 2] = (2*color+90)%256;
                imageData.data[pixelIndex + 3] = 255;
            }
        }
    }
    ctx.putImageData(imageData, 0, 0);
}
